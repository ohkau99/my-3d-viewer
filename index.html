function createFunctionAtlas(model) { removeFunctionAtlas(); const box = new THREE.Box3().setFromObject( model ); const size = box.getSize( new THREE.Vector3() ); const center = box.getCenter( new THREE.Vector3() ); functionAtlasGroup = new THREE.Group(); functionAtlasGroup.name = "BrainFunctionAtlas"; const atlasRegions = [ { name: "Motor Cortex", color: 0x22d3ee, position: [ 0.00, 0.10, 0.03 ], scale: [ 0.08, 0.10, 0.05 ], labelOffset: [ 0.00, 0.26, 0.02 ] }, { name: "Sensory Cortex", color: 0x4ade80, position: [ 0.00, 0.00, 0.05 ], scale: [ 0.08, 0.11, 0.05 ], labelOffset: [ 0.00, 0.02, 0.20 ] }, { name: "Language Area", color: 0xfacc15, position: [ 0.14, -0.04, 0.05 ], scale: [ 0.08, 0.08, 0.05 ], labelOffset: [ 0.24, -0.12, 0.02 ] }, { name: "Visual Cortex", color: 0xc084fc, position: [ -0.14, -0.04, -0.03 ], scale: [ 0.08, 0.08, 0.05 ], labelOffset: [ -0.24, -0.12, 0.02 ] } ]; atlasRegions.forEach( region => { const regionGroup = new THREE.Group(); const geometry = new THREE.SphereGeometry( 1, 48, 32 ); const material = new THREE.MeshPhysicalMaterial({ color: region.color, transparent: true, opacity: 0, roughness: 0.25, metalness: 0, transmission: 0.05, emissive: region.color, emissiveIntensity: 0.12, depthWrite: false, depthTest: true, side: THREE.DoubleSide }); material.userData.baseOpacity = 0.18; const mesh = new THREE.Mesh( geometry, material ); mesh.position.set( center.x + region.position[0] * size.x, center.y + region.position[1] * size.y, center.z + region.position[2] * size.z ); mesh.scale.set( region.scale[0] * size.x, region.scale[1] * size.y, region.scale[2] * size.z ); atlasRegionMaterials.push( material ); regionGroup.add( mesh ); const edgeGeometry = new THREE.SphereGeometry( 1, 24, 16 ); const edgeMaterial = new THREE.MeshBasicMaterial({ color: region.color, transparent: true, opacity: 0, wireframe: true, depthWrite: false, depthTest: true, side: THREE.DoubleSide }); edgeMaterial.userData.baseOpacity = 0.03; const edgeMesh = new THREE.Mesh( edgeGeometry, edgeMaterial ); edgeMesh.position.copy( mesh.position ); edgeMesh.scale.copy( mesh.scale ); edgeMesh.scale.multiplyScalar( 1.01 ); atlasRegionMaterials.push( edgeMaterial ); regionGroup.add( edgeMesh ); const label = createTextLabel( region.name, region.color, Math.max( size.x, size.y, size.z ) ); label.position.set( center.x + region.labelOffset[0] * size.x, center.y + region.labelOffset[1] * size.y, center.z + region.labelOffset[2] * size.z ); regionGroup.add( label ); functionAtlasGroup.add( regionGroup ); } ); functionAtlasGroup.visible = false; atlasVisible = false; atlasFadeValue = 0; atlasFadeTarget = 0; atlasButton.classList.remove( "active" ); atlasButton.textContent = "Brain Function Map"; scene.add( functionAtlasGroup ); }
